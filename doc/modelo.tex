}

\subsubsection*{Restricciones}

\modelRestrictionShouldBeIn{VehiculoEnUso.PATENTE}{Vehiculo.PATENTE}
\modelRestrictionShouldBeIn{VehiculoEnReparacion.PATENTE}{Vehiculo.PATENTE}
\modelRestrictionShouldBeInOr{Vehiculo.PATENTE}{VehiculoEnUso.PATENTE}{VehiculoEnReparacion.PATENTE}
\modelRestrictionShouldBeIn{Vehiculo.IDMODELO}{Modelo.IDMODELO}
\modelRestrictionShouldBeIn{Modelo.IDMARCA}{Marca.IDMARCA}
